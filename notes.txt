Purpose of file: Keeping track of files with separation of concerns. Directions etc
Backend
src/
├── config/
│   ├── database.ts          // Database connection & config  sequelise setup
│   └── passport.ts          // Passport JWT strategy setup
├── middleware/
│   ├── auth.ts              // Authentication middleware      reusable authentication "passport.authenticate('jwt', { session: false })"
│   └── errorHandler.ts      // Global error handler
├── controllers/
│   ├── taskController.ts    // All task-related logic
│   ├── shareController.ts   // All share-related logic
│   └── userController.ts    // User-related logic
├── routes/
│   ├── auth.ts              // Auth routes     defines endpoints, signup/login logic, handles requests & responses
│   ├── tasks.ts             // Task routes
│   ├── shares.ts            // Share routes
│   └── index.ts             // Combine all routes
├── services/
│   ├── taskService.ts       // Business logic for tasks everything task related
│   └── socketService.ts     // Socket.io logic  
├── models/                 //ORM
│   ├── index.ts
│   ├── User.ts
│   ├── Task.ts
│   └── Share.ts
├── types/
│   └── index.ts           //center for files to call functions
├── utils/
│   └── constants.ts         // JWT_SECRET, PORT, etc. centralises config values
└── app.ts                   // Slim entry point

Frontend
web-interface/
├── js/
│   ├── socket.js         // All Socket.IO event handling
│   ├── auth.js           // Login, logout, check user
│   ├── tasks.js          // Add, delete, toggle, load tasks
│   ├── categories.js     // Category select logic, new category input
│   ├── share.js          // Share modal logic
│   └── form.js           // Dynamic form behavior (text/image fields)
